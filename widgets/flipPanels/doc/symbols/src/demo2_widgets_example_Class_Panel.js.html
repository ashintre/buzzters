 17 /**

 19 */
 20 Panels.prototype.generatePanels = function(panels){
 21         this.panels = panels;
 22 	for(var i = 0; i < this.panels.length; i++){
 23 		var div = "<div id='" + this.panels[i].id + "' class='theme-content'>" + panels[i].html + "</div>";
 24 		document.getElementById('panel_container').innerHTML = document.getElementById('panel_container').innerHTML + div;
 25 	}

 28 /**
 29     @description Switch panels
 30     @param {integer} index Panel index to be displayed
 31 */
 32 Panels.prototype.switchPanels = function(index){
 33 	this.currentPanelIndex = index; // set current panel index to new index
 34         $(this.currentBtn).blur(); // reset current button information for panel switch
 35 	this.currentBtn = null; // reset current button information for panel switch
 36 	for(var i = 0; i < this.panels.length; i++){
 37 		if(i == this.currentPanelIndex){
 38 			$("#" + this.panels[i].id).show();
 39 		}
 40 		else{
 41 			$("#" + this.panels[i].id).hide();
 42 		}
 43 	}
 44 }
