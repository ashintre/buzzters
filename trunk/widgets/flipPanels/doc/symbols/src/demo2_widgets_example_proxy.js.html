  4 /**
  5 	@description Get Popular Digg News
  6 */
  7 function getDiggPopular(){
  8 
  9 	var proxyUrl = "/quercus-4.0.3/WidgetProxy.php?query=";
 10 	var diggAPI = "http://services.digg.com/1.0/endpoint?method=story.getPopular&count=3&type=json";
 11 
 12 	var req = proxyUrl + encodeURIComponent(diggAPI);
 13 	
 14 	$.getJSON(req, function(data){
 15 		if ((data !== null) &&
 16 	        (data !== undefined)){
 17 			var newHtml = "";
 18 			for(var i = 0; i < data.count; i++){
 19 				newHtml = newHtml 
 20                                     + "<p>"
 21                                     + data.stories[i].title
 22                                     + "</p>";
 23 			}
 24 			updatePanel(currentPanelIndex, newHtml);
 25 		}				
 26 	});
 27 }
